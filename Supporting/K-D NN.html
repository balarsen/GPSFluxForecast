for offset in tqdm.tqdm(range(36)):
    storm_v = dst['Dst'].iloc[dst_ind-25+offset:dst_ind+offset].values.reshape(1, -1)
    dist, ind = tree.query(storm_v, k=30) 

    dd = dst.iloc[dst_ind-25:dst_ind+71]
    XX = dd.index.to_pydatetime()
    ans = []
    for i in ind[0]:
        ans.append(dst.iloc[i:i+96].values.reshape(-1))

    ans = np.asarray(ans)
    perc = np.percentile(ans, (25, 50, 75), axis=0)
    mean = np.mean(ans, axis=0)
    plt.figure(figsize=(15,6))
    dd = dst.iloc[dst_ind-25+offset:dst_ind+71+offset]

    XX = dd.index.to_pydatetime()

    plt.plot(XX, dd.values.reshape(-1), c='r', lw=3, label='Data')
    plt.xlim((datetime.datetime(2010, 5, 2), datetime.datetime(2010, 5, 6,12)))
    plt.ylim((-150, 60))
    
    for ii, v in enumerate(ans):
        if ii == 0:
            plt.plot(XX, v, c='grey', alpha=0.3, label='Neighbor')
        else:
            plt.plot(XX, v, c='grey', alpha=0.3)


    plt.plot(XX, mean, c='g', lw=2, label='mean')
    plt.plot(XX, perc.T, c='orange', lw=3, label='IQR')
    plt.legend(loc='lower left')
    plt.xlabel('2010')
    plt.ylabel('Dst [nT]')
    plt.axvline(dd.index[25], c='k')
    plt.savefig('dst_plot_{:04}.png'.format(offset))
    plt.close('all')
